---
title: "database.conf"
---

指定插件所使用的数据库。

```hocon
# MariaDB 数据库配置
mariadb {
    # 数据库名称
    database="hyperzone_login"
    # ⚠️MariaDB JDBC驱动类名。通常不需要修改。
    driver-class-name="org.mariadb.jdbc.Driver"
    # MariaDB 服务器地址
    host=localhost
    # 额外的连接参数
    parameters="useSSL=false&characterEncoding=utf8"
    # 密码
    password=password
    # MariaDB 服务器端口
    port=3306
    # 用户名
    username=root
}
# MySQL 数据库配置
mysql {
    # 数据库名称
    database="hyperzone_login"
    # ⚠️MySQL JDBC驱动类名。通常不需要修改。
    driver-class-name="com.mysql.cj.jdbc.Driver"
    # MySQL 服务器地址
    host=localhost
    # ⚠️连接MySQL服务器的额外JDBC参数
    parameters="useSSL=false&serverTimezone=UTC&characterEncoding=utf8"
    # 密码
    password=password
    # MySQL 服务器端口
    port=3306
    # 用户名
    username=root
}
# 连接池配置
pool {
    # ⚠️连接超时时间（毫秒）
    connection-timeout=30000
    # ⚠️空闲连接超时时间（毫秒）
    idle-timeout=600000
    # ⚠️连接最大生命周期（毫秒）
    max-lifetime=1800000
    # ⚠️连接池最大连接数
    maximum-pool-size=10
    # ⚠️连接池最小空闲连接数
    minimum-idle=2
}
# SQLite 数据库配置
sqlite {
    # 数据库文件路径（相对于插件数据目录）
    path="data/hyperzone_login.db"
}
# 数据库表前缀
table-prefix="hz_"
# 数据库类型
#        支持的值: SQLITE, MYSQL, MARIADB
type=SQLITE
```
